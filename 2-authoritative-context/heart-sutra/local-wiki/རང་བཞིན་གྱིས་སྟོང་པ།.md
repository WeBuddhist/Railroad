_རང་བཞིན་གྱིས་སྟོང་པ་_ཞེས་པ་ནི་དབུ་མའི་ལྟ་བའི་གཙོ་བོ་ཞིག་སྟེ། ཆོས་ཐམས་ཅད་རང་གི་ངོ་བོར་མ་གྲུབ་པར་བསྟན་པའི་ཐ་སྙད་དོ། །

## སྒྲ་བཤད།

**རང་བཞིན་གྱིས་སྟོང་པ་**ཞེས་པ་རང་གི་ངོ་བོ་ཉིད་ཀྱིས་སྟོང་པ་ཡིན་པས་ན་དེ་ལྟར་བརྗོད་དོ།།

## ནང་དོན།

**ཤེས་རབ་སྙིང་པོའི་ནང་དུ་གསུངས་པ།** ཤེས་རབ་སྙིང་པོའི་མདོ་ནང་། "ཕུང་པོ་ལྔ་པོ་དེ་དག་ལ་ཡང་རང་བཞིན་གྱིས་སྟོང་པར་རྣམ་པར་ལྟའོ།"[[༡]](https://bo.wikipedia.org/wiki/%E0%BD%A2%E0%BD%84%E0%BC%8B%E0%BD%96%E0%BD%9E%E0%BD%B2%E0%BD%93%E0%BC%8B%E0%BD%82%E0%BE%B1%E0%BD%B2%E0%BD%A6%E0%BC%8B%E0%BD%A6%E0%BE%9F%E0%BD%BC%E0%BD%84%E0%BC%8B%E0%BD%94%E0%BC%8B#cite_note-1) ཞེས་གསུངས་པ་ལྟར། ཕུང་པོ་ལྔ་དང་ཆོས་ཐམས་ཅད་ཀྱང་རང་བཞིན་གྱིས་སྟོང་པར་གསུངས་འདུག ། དྲི་མེད་བཤེས་གཉེན་གྱིས་ཀྱང་། "དེ་དག་ཀྱང་ངོ་བོ་ཉིད་ཀྱིས་སྟོང་པ་ཞེས་འབྱུང་བ་འདིས་ནི་ཇི་ལྟ་བ་བཞིན་དུ་ཡོད་པར་བསྟན་ཏེ། །ངོ་བོ་ཉིད་ཀྱིས་སྟོང་པ་དང་། བྲལ་བ་དང་། བདག་མེད་པ་ཞེས་བྱ་བའི་དོན་ཏོ། །འདི་ཡང་མཚན་ཉིད་མེད་པ་དང་། མ་སྐྱེས་པ་ལ་སོགས་པ་མཚོན་པ་ཡིན་ནོ། །"[[༢]](https://bo.wikipedia.org/wiki/%E0%BD%A2%E0%BD%84%E0%BC%8B%E0%BD%96%E0%BD%9E%E0%BD%B2%E0%BD%93%E0%BC%8B%E0%BD%82%E0%BE%B1%E0%BD%B2%E0%BD%A6%E0%BC%8B%E0%BD%A6%E0%BE%9F%E0%BD%BC%E0%BD%84%E0%BC%8B%E0%BD%94%E0%BC%8B#cite_note-2) ཞེས་གསུངས་སོ། ། **གཟུགས་སྟོང་པའི་ཚུལ།** "གཟུགས་སྟོང་པའོ། སྟོང་པ་ཉིད་གཟུགས་སོ། གཟུགས་ལས་ཀྱང་སྟོང་པ་ཉིད་གཞན་མ་ཡིན། སྟོང་པ་ཉིད་ལས་ཀྱང་གཟུགས་གཞན་མ་ཡིན་ནོ།"[[༣]](https://bo.wikipedia.org/wiki/%E0%BD%A2%E0%BD%84%E0%BC%8B%E0%BD%96%E0%BD%9E%E0%BD%B2%E0%BD%93%E0%BC%8B%E0%BD%82%E0%BE%B1%E0%BD%B2%E0%BD%A6%E0%BC%8B%E0%BD%A6%E0%BE%9F%E0%BD%BC%E0%BD%84%E0%BC%8B%E0%BD%94%E0%BC%8B#cite_note-3) ཞེས་གསུངས་པས། གཟུགས་དང་སྟོང་པ་ཉིད་གཅིག་ཏུ་ཡིན་པར་བསྟན་ནོ། ། དྲི་མེད་བཤེས་གཉེན་གྱིས། "གཟུགས་སྟོང་པའོ་ཞེས་བཀའ་སྩལ་ཏེ། སྐབས་ཡིན་པའི་ཕྱིར་རང་གི་ངོ་བོ་ཉིད་ཀྱིས་ཞེས་འབྱུང་བ་དང་སྦྱར་རོ། །འདི་ལྟར་གཟུགས་ལ་སོགས་པར་སྣང་བ་ལ་དངོས་པོ་ཡིན་པ་དང་། བཏགས་པ་ཡིན་པར་སྣང་བ་རྣམས་རྣམ་པ་དེ་གཉིས་སུ་རྟོག་པར་བྱེད་དོ། །"[[༤]](https://bo.wikipedia.org/wiki/%E0%BD%A2%E0%BD%84%E0%BC%8B%E0%BD%96%E0%BD%9E%E0%BD%B2%E0%BD%93%E0%BC%8B%E0%BD%82%E0%BE%B1%E0%BD%B2%E0%BD%A6%E0%BC%8B%E0%BD%A6%E0%BE%9F%E0%BD%BC%E0%BD%84%E0%BC%8B%E0%BD%94%E0%BC%8B#cite_note-4) ཞེས་གསུངས་སོ། ། **ཆུ་ཟླའི་དཔེ།** "འདི་ནི་ཆུ་ཟླའི་དཔེ་ཤེས་པར་བྱ་སྟེ། དཔེར་ན་ཆུ་ནང་དུ་ཟླ་བའི་གཟུགས་བརྙན་སྣང་བ་ན། ཆུ་ནང་དུ་ཟླ་བ་སྣང་ཙམ་མ་འགག་པར་ཡོད་པ་དང་། ཟླ་བ་སྣང་བའི་དུས་ནས་ཆུ་ནང་དུ་ཟླ་བར་ཡོད་མ་མྱོང་བ་དང་ཟླ་བར་སྣང་བ་དང་། ཟླ་བའི་རང་གི་ངོ་བོར་ཡོད་མ་མྱོང་པ་ཐ་མི་དད་པ་དང་། ཟླ་བའི་མཚན་ཉིད་ཤེས་ནས་རང་སར་གྲོལ་བའོ། །དཔེ་དེ་བཞིན་དུ་གཟུགས་སུ་སྣང་བའི་དུས་ཉིད་ནས་གཟུགས་ཀྱི་རང་གི་ངོ་བོས་སྟོང་པ་སྟེ། འདི་ནི་གཟུགས་སྟོང་པའོ་ཞེས་པའི་དོན་ནོ། །"[[༥]](https://bo.wikipedia.org/wiki/%E0%BD%A2%E0%BD%84%E0%BC%8B%E0%BD%96%E0%BD%9E%E0%BD%B2%E0%BD%93%E0%BC%8B%E0%BD%82%E0%BE%B1%E0%BD%B2%E0%BD%A6%E0%BC%8B%E0%BD%A6%E0%BE%9F%E0%BD%BC%E0%BD%84%E0%BC%8B%E0%BD%94%E0%BC%8B#cite_note-5)

## མཚན་ཉིད།

རང་བཞིན་གྱིས་སྟོང་པ་ཞེས་པ་ནི་རང་གི་ངོ་བོས་མ་གྲུབ་པ་སྟེ། "དཔེ་ཆུ་ནང་དུ་ཟླ་བར་སྣང་ཡང་ཟླ་བའི་ངོ་བོར་མ་གྲུབ་པ་བཞིན་ནོ།"[[༦]](https://bo.wikipedia.org/wiki/%E0%BD%A2%E0%BD%84%E0%BC%8B%E0%BD%96%E0%BD%9E%E0%BD%B2%E0%BD%93%E0%BC%8B%E0%BD%82%E0%BE%B1%E0%BD%B2%E0%BD%A6%E0%BC%8B%E0%BD%A6%E0%BE%9F%E0%BD%BC%E0%BD%84%E0%BC%8B%E0%BD%94%E0%BC%8B#cite_note-6) ཞེས་བཤད་དོ། ། དྲི་མེད་བཤེས་གཉེན་གྱིས། "གང་དེ་དག་བ་ལང་རྫི་དང་བུད་མེད་ཡན་ཆད་ལ་གྲགས་པ་དངོས་སུ་གསལ་བར་སྣང་བའི་གཟུགས་དེ་ཉིད་དྲི་ཟའི་གྲོང་ཁྱེར་ལྟར་ངོ་བོ་ཉིད་ཀྱིས་སྟོང་པ་ཡིན་ཏེ། གཟུགས་སྣང་བ་འདི་ལ་རང་གི་ངོ་བོ་འགའ་ཡང་མེད་དོ།"[[༧]](https://bo.wikipedia.org/wiki/%E0%BD%A2%E0%BD%84%E0%BC%8B%E0%BD%96%E0%BD%9E%E0%BD%B2%E0%BD%93%E0%BC%8B%E0%BD%82%E0%BE%B1%E0%BD%B2%E0%BD%A6%E0%BC%8B%E0%BD%A6%E0%BE%9F%E0%BD%BC%E0%BD%84%E0%BC%8B%E0%BD%94%E0%BC%8B#cite_note-7) ཞེས་གསུངས་སོ། །

## སྟོང་ཚུལ།

"སྤྱིར་སྟོང་ཚུལ་ལྔ་ཡོད་དེ། འོ་མའི་དུས་ན་ཞོ་མེད་པ་ནི་སྔ་ནས་མེད་པའི་སྟོང་པའོ། །ཞོའི་དུས་ན་འོ་མ་མེད་པ་ནི་ཞིག་ནས་མེད་པའི་སྟོང་པའོ། །ལྟོ་ལྡིར་བ་མེད་པའི་ས་ཕྱོགས་ན་བུམ་པ་མེད་པ་ནི་ཤིན་ཏུ་མེད་པའི་སྟོང་པའོ། །བ་ལང་ལ་རྟ་མེད་པ་ནི་གཅིག་ལ་གཅིག་ཤོས་མེད་པའི་སྟོང་པའོ། །ཐོ་ཡོར་ལ་མིར་འཁྲུལ་པའི་ཚེ་དེ་ལ་མི་མེད་པ་ནི་ངོ་བོ་ཉིད་ཀྱིས་སྟོང་པའོ། །སྐབས་འདིའི་སྟོང་ཚུལ་ཐ་མའི་སྟོང་ཚུལ་དང་གནད་གཅིག་པས།"[[༨]](https://bo.wikipedia.org/wiki/%E0%BD%A2%E0%BD%84%E0%BC%8B%E0%BD%96%E0%BD%9E%E0%BD%B2%E0%BD%93%E0%BC%8B%E0%BD%82%E0%BE%B1%E0%BD%B2%E0%BD%A6%E0%BC%8B%E0%BD%A6%E0%BE%9F%E0%BD%BC%E0%BD%84%E0%BC%8B%E0%BD%94%E0%BC%8B#cite_note-8)

## ལུགས་སྣ་ཚོགས་ཀྱི་འགྲེལ་བཤད།

**སེམས་ཙམ་པའི་ལུགས།** སེམས་ཙམ་པ་ལྟར་ན། "གཟུགས་སུ་སྣང་བ་འདི་གཟུང་འཛིན་གཉིས་ཀྱིས་སྟོང་པའི་རང་རིག་གོ། །རང་རིག་དེ་ནི་གཟུགས་སུ་སྣང་བའོ། །གཟུགས་སྣང་བ་ལས་རང་རིག་གཞན་མ་ཡིན། རང་རིག་ལས་ཀྱང་གཟུགས་སུ་སྣང་བ་གཞན་མ་ཡིན་ནོ།"[[༩]](https://bo.wikipedia.org/wiki/%E0%BD%A2%E0%BD%84%E0%BC%8B%E0%BD%96%E0%BD%9E%E0%BD%B2%E0%BD%93%E0%BC%8B%E0%BD%82%E0%BE%B1%E0%BD%B2%E0%BD%A6%E0%BC%8B%E0%BD%A6%E0%BE%9F%E0%BD%BC%E0%BD%84%E0%BC%8B%E0%BD%94%E0%BC%8B#cite_note-9) ཞེས་པའི་དོན་ཅན་དོ། ། **ཐལ་འགྱུར་བའི་ལུགས།** ཐལ་འགྱུར་པ་ལྟར་ན། "གཟུགས་སུ་སྣང་བས་ན་མེད་པའོ། །མེད་པས་ན་གཟུགས་སུ་སྣང་བའོ། །གཟུགས་སུ་སྣང་བ་འདི་ག་ལ་རང་མེད་པ་ཡིན་གྱི་དེའི་མེད་པ་དེ་ལས་གཞན་དུ་བཙལ་དགོས་པ་མ་ཡིན། དེ་མེད་བཞིན་དུ་གཟུགས་སུ་སྣང་བ་ཡིན་པས་དེ་ལས་གཟུགས་ལོགས་སུ་ཡོད་པ་མ་ཡིན།"[[༡༠]](https://bo.wikipedia.org/wiki/%E0%BD%A2%E0%BD%84%E0%BC%8B%E0%BD%96%E0%BD%9E%E0%BD%B2%E0%BD%93%E0%BC%8B%E0%BD%82%E0%BE%B1%E0%BD%B2%E0%BD%A6%E0%BC%8B%E0%BD%A6%E0%BE%9F%E0%BD%BC%E0%BD%84%E0%BC%8B%E0%BD%94%E0%BC%8B#cite_note-10) ཞེས་པའི་དོན་ནོ། །

## སྒོམ་ཚུལ།

**མཉམ་བཞག་སྒོམ་ཚུལ།** "སྟོང་པ་ཉིད་ཡང་དག་པར་རྗེས་སུ་མཐོང་པ་ལ་གཟུགས་ཡང་དག་པར་རྗེས་སུ་བལྟར་མེད་ཅེས་པའི་དགོངས་པ་ཡིན་པས། དེས་ན་མཉམ་བཞག་གི་གཟིགས་ངོ་ལ་ཕུང་པོ་ལྔ་མི་སྣང་བ་ནི། གཟུགས་མེད་ཅེས་པ་ནས་རྣམ་པར་ཤེས་པ་མེད་ཅེས་པའི་བར་རོ།"[[༡༡]](https://bo.wikipedia.org/wiki/%E0%BD%A2%E0%BD%84%E0%BC%8B%E0%BD%96%E0%BD%9E%E0%BD%B2%E0%BD%93%E0%BC%8B%E0%BD%82%E0%BE%B1%E0%BD%B2%E0%BD%A6%E0%BC%8B%E0%BD%A6%E0%BE%9F%E0%BD%BC%E0%BD%84%E0%BC%8B%E0%BD%94%E0%BC%8B#cite_note-11) དྲི་མེད་བཤེས་གཉེན་གྱིས། "དེ་ལྟ་བས་ན་དེའི་ཚེ་སྟོང་པ་ཉིད་ཡང་དག་པར་རྗེས་སུ་མཐོང་བ་ལ་གཟུགས་ཡང་དག་པར་རྗེས་སུ་བལྟར་མེད་ཅེས་འབྱུང་བར་སྦྱར་ཏེ། ཆོས་ཐམས་ཅད་མི་མཐོང་བ་ཉིད་ནི་སྟོང་པ་ཉིད་མཐོང་བ་ཡིན་ནོ།"[[༡༢]](https://bo.wikipedia.org/wiki/%E0%BD%A2%E0%BD%84%E0%BC%8B%E0%BD%96%E0%BD%9E%E0%BD%B2%E0%BD%93%E0%BC%8B%E0%BD%82%E0%BE%B1%E0%BD%B2%E0%BD%A6%E0%BC%8B%E0%BD%A6%E0%BE%9F%E0%BD%BC%E0%BD%84%E0%BC%8B%E0%BD%94%E0%BC%8B#cite_note-12) ཞེས་གསུངས་སོ། །

## དགོས་དོན།

**མཐའ་གཉིས་སྤོང་བ།** "སོ་སོ་སྐྱེ་བོ་རྣམས་གཟུགས་སོགས་ལ་ཆགས་ཏེ་འཁོར་བའི་མཐར་ལྟུང་བ་འགོག་ཆེད་དུ་གཟུགས་སྟོང་པའོ་ཞེས་གདམས་པ་ན། ཡང་འགའ་ཞིག་གིས་སྟོང་པ་ལ་ཆགས་ཏེ་ཞི་མཐར་ལྟུང་བ་འགོག་ཆེད་དུ་སྟོང་པ་ཉིད་གཟུགས་སོ་ཞེས་གདམས་པ་ཡིན་ཏེ། དཔེར་ན་གཡས་ཕྱོགས་ན་ཚེར་མ་དང་གཡོན་ན་ངམ་གྲོག་ཡོད་པའི་ལམ་དོག་མོའི་ནང་དུ་མིག་མེད་པ་ཞིག་འགྲོ་པ་ན། མིག་ལྡན་ཞིག་གིས་ཚེར་མ་ཡོད་དོ་བྱས་ན་ངམ་གྲོག་ཏུ་ལྷུང་ཉེན་ཡོད། ངམ་གྲོག་ཡོད་དོ་བྱས་ན་ཚེར་མར་ལྷུང་ཉེན་ཡོད་པའི་མཐའ་གཉིས་སུ་མི་ལྷུང་བའི་ཆེད་དུ་དབུ་མའི་ལམ་བསྟན་པ་ལྟ་བུའོ།"[[༡༣]](https://bo.wikipedia.org/wiki/%E0%BD%A2%E0%BD%84%E0%BC%8B%E0%BD%96%E0%BD%9E%E0%BD%B2%E0%BD%93%E0%BC%8B%E0%BD%82%E0%BE%B1%E0%BD%B2%E0%BD%A6%E0%BC%8B%E0%BD%A6%E0%BE%9F%E0%BD%BC%E0%BD%84%E0%BC%8B%E0%BD%94%E0%BC%8B#cite_note-13) **མྱ་ངན་ལས་འདས་པའི་ཐབས།** "བཅོམ་ལྡན་འདས་ཀྱི་དགོངས་པ་ནི་སེམས་ཅན་མྱ་ངན་ལས་འདས་པའི་ཐབས་འབའ་ཞིག་དགོངས་ཤིང་། མྱ་ངན་ལས་འདའ་བར་བྱེད་པའི་ཐབས་ནི་སྟོང་ཉིད་རྟོགས་པའི་ཤེས་རབ་ཡིན་པས་དེ་ལ་སྙིང་པོ་ཞེས་བྱ་ཞིང་།"[[༡༤]](https://bo.wikipedia.org/wiki/%E0%BD%A2%E0%BD%84%E0%BC%8B%E0%BD%96%E0%BD%9E%E0%BD%B2%E0%BD%93%E0%BC%8B%E0%BD%82%E0%BE%B1%E0%BD%B2%E0%BD%A6%E0%BC%8B%E0%BD%A6%E0%BE%9F%E0%BD%BC%E0%BD%84%E0%BC%8B%E0%BD%94%E0%BC%8B#cite_note-14) ཞེས་གསུངས་སོ། ། དྲི་མེད་བཤེས་གཉེན་གྱིས། "དེ་བས་ན་རང་གི་ངོ་བོ་སྟོང་པ་ཉིད་འདི་ལ་གཟུགས་སོ་ཞེས་བྱ་སྟེ། འདིས་ནི་བྱེ་བྲག་ཏུ་སྨྲ་བ་དང་། སྤྱོད་པ་པ་ལ་སོགས་པ་འདོད་པའི་གཟུགས་རང་གི་ངོ་བོ་སྟོང་པ་ཉིད་དུ་བསྟན་པ་ཡིན་ནོ།"[[༡༥]](https://bo.wikipedia.org/wiki/%E0%BD%A2%E0%BD%84%E0%BC%8B%E0%BD%96%E0%BD%9E%E0%BD%B2%E0%BD%93%E0%BC%8B%E0%BD%82%E0%BE%B1%E0%BD%B2%E0%BD%A6%E0%BC%8B%E0%BD%A6%E0%BE%9F%E0%BD%BC%E0%BD%84%E0%BC%8B%E0%BD%94%E0%BC%8B#cite_note-15) ཞེས་གསུངས་སོ། །
